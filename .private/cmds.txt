### SERVER
# create server folder and inside server folder:
python -m venv .venv
source .venv/bin/activate
pip install Flask
pip install flask-cors

# run this in server folder to start flask app
python app.py


### CLIENT 

# run this command in the root folder to create client folder
npm create vite@latest client -- --template react

# run this inside client folder
npm install vite (so the vite command works)
